async function generatePDF() { const { jsPDF } = window.jspdf; // 1. Setup Data const m = parseFloat(document.getElementById('module-sel').value); const z1 = parseInt(document.getElementById('z1').value); const z2 = parseInt(document.getElementById('z2').value); const p = document.getElementById('in-p').value; const n = document.getElementById('in-n').value; const sigma0 = document.getElementById('material-stress').value; const date = new Date().toLocaleDateString(); document.getElementById('pdf-date').innerText = date; document.getElementById('pdf-max-stress-val').innerText = document.getElementById('max-stress-label').innerText; // 2. Draw 2D Work (Schematic) const can = document.getElementById('pdf-2d-canvas'); const ctx = can.getContext('2d'); ctx.clearRect(0, 0, can.width, can.height); const scale = 1.5; const cx = 100, cy = 125; const r1 = (z1 * m) / 2 * scale; const r2 = (z2 * m) / 2 * scale; // Draw Pinion ctx.beginPath(); ctx.arc(cx, cy, r1, 0, Math.PI * 2); ctx.strokeStyle = '#3b82f6'; ctx.lineWidth = 2; ctx.stroke(); ctx.fillText(`Pinion (Z=${z1})`, cx - 20, cy - r1 - 5); // Draw Gear ctx.beginPath(); ctx.arc(cx + r1 + r2, cy, r2, 0, Math.PI * 2); ctx.strokeStyle = '#1e293b'; ctx.stroke(); ctx.fillText(`Gear (Z=${z2})`, cx + r1 + r2 - 20, cy - r2 - 5); // Line of Centers ctx.setLineDash([5, 5]); ctx.beginPath(); ctx.moveTo(cx, cy); ctx.lineTo(cx + r1 + r2, cy); ctx.stroke(); ctx.setLineDash([]); // 3. Populate Inputs document.getElementById('pdf-inputs-grid').innerHTML = ` Power: ${p} kWSpeed: ${n} RPM Module: ${m} mmZ1/Z2: ${z1}/${z2} Stress (σo): ${sigma0} MPaTooth: ${document.getElementById('tooth-system').value}° `; // 4. Solve Calculations for PDF // Re-calculating locally for explicit string substitution const v = (Math.PI * (z1 * m) * n) / 60000; const ft = (1000 * parseFloat(p)) / v; const cs = serviceFactorTable[document.getElementById('load-type').value][document.getElementById('service-type').value]; const ftEff = ft * cs; document.getElementById('pdf-calculations').innerHTML = ` STEP 1: VELOCITY & LOAD
V = (π × d1 × n1) / 60000 = (π × ${(z1*m)} × ${n}) / 60000 = ${v.toFixed(2)} m/s
Ft = (1000 × P) / V = (1000 × ${p}) / ${v.toFixed(2)} = ${ft.toFixed(2)} N
Ft(eff) = Ft × Cs = ${ft.toFixed(2)} × ${cs} = ${ftEff.toFixed(2)} N

STEP 2: BEAM STRENGTH (LEWIS)
Fs = σo × b × π × m × y
Fs = ${sigma0} × ${(10*m)} × π × ${m} × ${document.getElementById('step-output').innerText.match(/Form Factor \(y\): ([\d.]+)/)[1]}
Result: ${document.getElementById('step-output').innerText.match(/Beam Strength \(Fs\): ([\d.]+)/)[1]} N

STEP 3: DYNAMIC CHECK
Using Buckingham Equation for Dynamic Load (Fd)...
Calculated Fd: ${document.getElementById('step-output').innerText.match(/Dynamic Load \(Fd\): ([\d.]+)/)[1]} N
FOS (Fs / Fd) = ${document.getElementById('step-output').innerText.match(/Factor of Safety: ([\d.]+)/)[1]} `; // 5. Render to PDF const canvas = await html2canvas(document.getElementById('pdf-template'), { scale: 2 }); const imgData = canvas.toDataURL('image/png'); const doc = new jsPDF('p', 'mm', 'a4'); const imgProps = doc.getImageProperties(imgData); const pdfWidth = doc.internal.pageSize.getWidth() - 20; const pdfHeight = (imgProps.height * pdfWidth) / imgProps.width; doc.addImage(imgData, 'PNG', 10, 10, pdfWidth, pdfHeight); doc.save(`Gear_Analysis_${z1}T_${z2}T.pdf`); }
